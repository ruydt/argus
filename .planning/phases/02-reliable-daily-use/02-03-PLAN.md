// startWALCheckpoint runs PRAGMA wal_checkpoint(PASSIVE) periodically to prevent
// unbounded WAL file growth from long-lived SSE read connections.
func startWALCheckpoint(ctx context.Context, db *sql.DB, interval time.Duration) {
    go func() {
        t := time.NewTicker(interval)
        defer t.Stop()
        for {
            select {
            case <-t.C:
                if _, err := db.ExecContext(ctx, `PRAGMA wal_checkpoint(PASSIVE)`); err != nil {

}

| SSE client → graceful shutdown | Open SSE connections can block Shutdown() forever without a finite context |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Denial of Service | HTTP server (no ReadHeaderTimeout) | mitigate | Set ReadHeaderTimeout: 5*time.Second — prevents Slowloris-style slow header attacks on the hook ingestion endpoint |
| T-02-03-02 | Denial of Service | graceful shutdown with context.Background() | mitigate | Replace with context.WithTimeout(15s) — ensures process terminates within 15s even with active SSE clients |
| T-02-03-03 | Denial of Service | WAL file unbounded growth | mitigate | startWALCheckpoint every 5 min with PASSIVE mode — non-blocking, prevents WAL from growing indefinitely due to long-lived SSE reads |
